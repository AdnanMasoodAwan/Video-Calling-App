47  bool operator==(const StreamLayerConfigInternal& rhs) const {
48  return dimensions == rhs.dimensions && bitrate_kbps == rhs.bitrate_kbps && framerate == rhs.framerate && enable == rhs.enable;
49  }

55  return *this;
56  }

58  void reset() {
